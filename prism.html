            import java.util.Scanner;

            public class Life {
            
                @Override @Bind("One")
                public void show(boolean[][] grid){
                    String s = "";
                    for(boolean[] row : grid){
                        for(boolean val : row)
                            if(val)
                                s += "*";
                            else
                                s += ".";
                        s += "\n";
                    }
                    System.out.println(s);
                }
            }
